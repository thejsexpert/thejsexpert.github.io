    Best Approach RTK

    Revolt 1

    src/
 redux/
     store/
        index.js        // Configure and export the Redux store
     middleware/
        customMiddleware.js // Your custom middleware (if any)
     blog/
        api.js          // API calls using Axios
        slice.js        // Redux slice for blog state
        services.js     // Business logic and service functions
     index.js            // Optional: combine all slices if needed

    Breakdown of Each File

    • Use this file to define your Axios instance and API calls.

    • Example:

      import axios from 'axios';

const api = axios.create({
    baseURL: 'https://your-api-url.com',
});

export const fetchBlogs = () => api.get('/blogs');
export const createBlog = (blogData) => api.post('/blogs', blogData);
    • This file contains the Redux slice for managing the blog state.

    • Example:

      import { createSlice } from '@reduxjs/toolkit';

const blogSlice = createSlice({
    name: 'blog',
    initialState: {
        blogs: [],
        loading: false,
        error: null,
    },
    reducers: {
        setBlogs: (state, action) => {
            state.blogs = action.payload;
        },
        setLoading: (state, action) => {
            state.loading = action.payload;
        },
        setError: (state, action) => {
            state.error = action.payload;
        },
    },
});

export const { setBlogs, setLoading, setError } = blogSlice.actions;
export default blogSlice.reducer;
    • Handle the business logic and asynchronous operations here, like dispatching actions.

    • Example:

      import { fetchBlogs } from './api';
import { setBlogs, setLoading, setError } from './slice';

export const loadBlogs = () => async (dispatch) => {
    dispatch(setLoading(true));
    try {
        const response = await fetchBlogs();
        dispatch(setBlogs(response.data));
    } catch (error) {
        dispatch(setError(error.message));
    } finally {
        dispatch(setLoading(false));
    }
};

    import React, { useEffect } from "react";
import { useDispatch, useSelector } from "react-redux";
import { loadBlogs } from "../redux/blog/services";

const BlogList = () => {
    const dispatch = useDispatch();
    const blogs = useSelector((state) => state.blog.blogs);
    const loading = useSelector((state) => state.blog.loading);
    const error = useSelector((state) => state.blog.error);

    useEffect(() => {
        dispatch(loadBlogs());
    }, [dispatch]);

    if (loading) return <div>Loading...</div>;
    if (error) return <div>Error: {error}</div>;

    return (
        <ul>
            {blogs.map((blog) => (
                <li key={blog.id}>{blog.title}</li>
            ))}
        </ul>
    );
};

export default BlogList;

    Revolt 2

    src/
 redux/
     store/
        index.js             // Configure and export the Redux store
     middleware/
        customMiddleware.js   // Your custom middleware (if any)
     features/
        blogs/
           api.js            // API calls for blogs
           slice.js          // Redux slice for blogs
           services.js       // Business logic for blogs
        users/
           api.js            // API calls for users
           slice.js          // Redux slice for users
           services.js       // Business logic for users
        comments/
           api.js            // API calls for comments
           slice.js          // Redux slice for comments
           services.js       // Business logic for comments
        ...                   // Other feature folders
     index.js                  // Optional: combine all slices if needed

    1. features/ Folder:

      • This is a dedicated folder for all your feature slices. Each feature (like blogs, users, comments, etc.) gets its own subfolder.

    2. Feature Structure:

      Inside each feature folder, you maintain the same structure:

      • api.js: Contains API calls relevant to that feature.
      • slice.js: Contains the Redux slice for managing that feature's state.
      • services.js: Contains the business logic and any async actions for that feature.
    • Scalability: You can easily add new features without cluttering the Redux folder. Each feature is self-contained.
    • Maintainability: It's easier to find related files for a specific feature. Developers can navigate directly to the feature they are working on.
    • Consistency: Following a consistent structure across all feature folders simplifies understanding the codebase.

    Updated api for crud

    import axios from "axios";

const apiClient = axios.create({
    baseURL: "https://jsonplaceholder.typicode.com",
});

const blogsApi = {
    fetchBlogs: (signal) => apiClient.get("/posts", { cancelToken: signal }),
    fetchBlogById: (id, signal) =>
        apiClient.get(`/posts/${id}`, { cancelToken: signal }),
    createBlog: (blogData) => apiClient.post("/posts", blogData),
    updateBlog: (id, blogData) => apiClient.put(`/posts/${id}`, blogData),
    deleteBlog: (id) => apiClient.delete(`/posts/${id}`),
};

export default blogsApi;

    import { createAsyncThunk } from "@reduxjs/toolkit";
import blogsApi from "./api"; // Import the API functions

export const fetchBlogs = createAsyncThunk(
    "blog/fetchBlogs",
    async (signal, { rejectWithValue }) => {
        try {
            const response = await blogsApi.fetchBlogs(signal);
            return response.data;
        } catch (err) {
            if (axios.isCancel(err)) {
                console.log("Request canceled:", err.message);
                return rejectWithValue("Request was canceled");
            }
            return rejectWithValue(err.message);
        }
    }
);

// Additional service functions for other CRUD operations
export const createBlog = createAsyncThunk(
    "blog/createBlog",
    async (blogData, { rejectWithValue }) => {
        try {
            const response = await blogsApi.createBlog(blogData);
            return response.data;
        } catch (err) {
            return rejectWithValue(err.message);
        }
    }
);

// Add similar functions for updateBlog and deleteBlog
    • Modularity: Each function in api.js handles a specific API call, making it clear and easy to manage.
    • Reusability: The blogsApi object can be reused across different service functions or even in different components if necessary.
    • Clarity: It's clear which API calls are available for the blog feature, making it easier for other developers (or future you) to understand the code.

    Decomposition

    import { createSlice, createAsyncThunk } from "@reduxjs/toolkit";
import axios from "axios";

export const fetchBlogs = createAsyncThunk(
  "blog/fetchBlogs",
  async (signal, { rejectWithValue }) => {
    try {
      const response = await axios.get(
        "https://jsonplaceholder.typicode.com/posts",
        {
          cancelToken: signal,
        }
      );
      return response.data;
    } catch (err) {
      if (axios.isCancel(err)) {
        console.log("Request canceled:", err.message);
        return rejectWithValue("Request was canceled");
      }
      return rejectWithValue(err.message);
    }
  }
);

const blogSlice = createSlice({
  name: "blog",
  initialState: {
    blogs: [],
    status: "idle",
    error: null,
  },
  extraReducers: (builder) => {
    builder
      .addCase(fetchBlogs.pending, (state) => {
        state.status = "loading";
      })
      .addCase(fetchBlogs.fulfilled, (state, action) => {
        state.status = "succeeded";
        state.blogs = action.payload;
        state.error = null;
      })
      .addCase(fetchBlogs.rejected, (state, action) => {
        state.status = "failed";
        state.error = action.payload;
      });
  },
});

export default blogSlice.reducer;

    To decompose your Redux slice code into multiple files while keeping your project organized, you'll want to separate your concerns into three distinct files: api.js, slice.js, and services.js. Heres how you can structure each of them:

    This file is responsible for making API requests. It contains functions that handle communication with your backend or external API. Youll define your Axios calls here.

    import axios from "axios";

const apiClient = axios.create({
    baseURL: "https://jsonplaceholder.typicode.com",
});

export const fetchBlogsApi = (signal) => {
    return apiClient.get("/posts", {
        cancelToken: signal,
    });
};

    This file contains your Redux slice. It defines the state structure, reducers, and actions related to the blog feature.

    import { createSlice } from "@reduxjs/toolkit";
import { fetchBlogs } from "./services"; // Import the async thunk from services

const blogSlice = createSlice({
    name: "blog",
    initialState: {
        blogs: [],
        status: "idle",
        error: null,
    },
    extraReducers: (builder) => {
        builder
            .addCase(fetchBlogs.pending, (state) => {
                state.status = "loading";
            })
            .addCase(fetchBlogs.fulfilled, (state, action) => {
                state.status = "succeeded";
                state.blogs = action.payload;
                state.error = null;
            })
            .addCase(fetchBlogs.rejected, (state, action) => {
                state.status = "failed";
                state.error = action.payload;
            });
    },
});

export const { actions, reducer } = blogSlice;
export default reducer;

    This file contains your asynchronous logic, such as createAsyncThunk functions. It uses the API functions defined in api.js to make requests.

    import { createAsyncThunk } from "@reduxjs/toolkit";
import { fetchBlogsApi } from "./api"; // Import the API function
import axios from "axios";

export const fetchBlogs = createAsyncThunk(
    "blog/fetchBlogs",
    async (signal, { rejectWithValue }) => {
        try {
            const response = await fetchBlogsApi(signal);
            return response.data;
        } catch (err) {
            if (axios.isCancel(err)) {
                console.log("Request canceled:", err.message);
                return rejectWithValue("Request was canceled");
            }
            return rejectWithValue(err.message);
        }
    }
);

    After splitting the functionality, your blogs folder structure will look like this:

    src/
 redux/
     features/
        blogs/
           api.js         // API requests
           slice.js       // Redux slice
           services.js     // Async thunks
     store/
         index.js           // Store configuration
    • api.js: Contains API call functions (using Axios).
    • slice.js: Contains the Redux slice definition, including initial state and reducers.
    • services.js: Contains asynchronous logic, such as thunks that use the API functions to fetch data.

const blogApiClient = axios.create({

// User API functions
const usersApi = {

    Approach 2: Single API Service with Dynamic Base URL

    import axios from "axios";

const apiClient = axios.create();

const setBaseURL = (type) => {
    switch (type) {
        case "blog":
            apiClient.defaults.baseURL = "https://jsonplaceholder.typicode.com";
            break;
        case "user":
            apiClient.defaults.baseURL = "https://jsonplaceholder.typicode.com/users";
            break;
        default:
            throw new Error("Unknown API type");
    }
};

const api = {
    fetchBlogs: (signal) => {
        setBaseURL("blog");
        return apiClient.get("/posts", { cancelToken: signal });
    },
    fetchUsers: () => {
        setBaseURL("user");
        return apiClient.get("/");
    },
    // Add other CRUD operations as needed
};

export default api;

    import React, { useEffect, useState } from "react";
import blogsApi from "./api"; // Adjust the import path as needed

const BlogList = () => {
    const [blogs, setBlogs] = useState([]);
    const [loading, setLoading] = useState(true);

    Example: Filtering Query Parameters in api.js

    import axios from "axios";

const apiClient = axios.create({
    baseURL: "https://jsonplaceholder.typicode.com",
});

// Helper function to filter and validate query parameters
const filterParams = (params) => {
    const filtered = {};
    if (params) {
        if (params.userId) {
            filtered.userId = params.userId; // Example filter
        }
        // Add more filtering logic as needed
    }
    return filtered;
};

const blogsApi = {
    fetchBlogs: (signal, params) => {
        const filteredParams = filterParams(params); // Filter params before sending
        return apiClient.get("/posts", {
            cancelToken: signal,
            params: filteredParams,
        });
    },
    fetchBlogById: (id, signal) =>
        apiClient.get(`/posts/${id}`, { cancelToken: signal }),
    createBlog: (blogData) => apiClient.post("/posts", blogData),
    updateBlog: (id, blogData) => apiClient.put(`/posts/${id}`, blogData),
    deleteBlog: (id) => apiClient.delete(`/posts/${id}`),
};

export default blogsApi;

    query-string

    The query-string package can be very useful when dealing with query parameters in URLs, especially if you need to handle complex scenarios like:

    1. Nested Objects: When you want to serialize nested objects as query parameters.
    2. Array Parameters: When you need to handle arrays and want them to be formatted in a specific way in the query string.
    3. URL Encoding: Automatically encoding and decoding query parameters can help avoid issues with special characters.
    When to Use query-string

    You might consider using the query-string package in the following scenarios:

    1. Complex Query Parameters: If you need to construct complex query strings, such as nested objects or arrays.

      import queryString from "query-string";

const params = {
    userId: 1,
    tags: ["react", "javascript"],
    filters: {
        sortBy: "date",
        order: "desc",
    },
};

const query = queryString.stringify(params);
// Result: "userId=1&tags[]=react&tags[]=javascript&filters[sortBy]=date&filters[order]=desc"

    2. Parsing Query Strings: If you need to parse incoming query strings from URLs into a usable object format.

      const parsed = queryString.parse(location.search);
// Example: location.search is "?userId=1&tags[]=react"
// Result: { userId: '1', tags: ['react'] }

    3. Avoiding Manual Serialization: If you find yourself frequently needing to manually construct query strings, using query-string can simplify your code.

    Heres how you might incorporate query-string into your existing blogsApi:

    1. Install the Package

      npm install query-string

    2. Update Your API Calls

      Update the fetchBlogs method to use query-string for parameter serialization:

      import axios from "axios";
import queryString from "query-string";

const apiClient = axios.create({
    baseURL: "https://jsonplaceholder.typicode.com",
});

const blogsApi = {
    fetchBlogs: (signal, params) => {
        const filteredParams = filterParams(params); // Assume you still filter params here
        const query = queryString.stringify(filteredParams); // Serialize query params
        return apiClient.get(`/posts?${query}`, { cancelToken: signal });
    },
    fetchBlogById: (id, signal) =>
        apiClient.get(`/posts/${id}`, { cancelToken: signal }),
    createBlog: (blogData) => apiClient.post("/posts", blogData),
    updateBlog: (id, blogData) => apiClient.put(`/posts/${id}`, blogData),
    deleteBlog: (id) => apiClient.delete(`/posts/${id}`),
};

export default blogsApi;
    • Use query-string when dealing with complex query parameters, such as nested objects or arrays.
    • It simplifies the process of constructing and parsing query strings, especially in more advanced scenarios.
    • If your API calls are simple (e.g., just basic key-value pairs), you might not need it, but it can still be a good addition for clarity and maintainability.
