    debounce

    npm install lodash

    jsx
    import React, { useState, useEffect } from "react";
import { Select } from "antd";
import debounce from "lodash/debounce";

const { Option } = Select;

const DebouncedSelect = () => {
  const [searchTerm, setSearchTerm] = useState("");

  const debouncedSearch = debounce((value) => {
    // Your search logic here, e.g., making an API call
    console.log("Searching for:", value);
  }, 300); // 300ms debounce delay

  useEffect(() => {
    debouncedSearch(searchTerm);
  }, [searchTerm, debouncedSearch]);

  const handleSearch = (value) => {
    setSearchTerm(value);
  };

  return (
    <Select
      showSearch
      onSearch={handleSearch}
      style={{ width: 200 }}
      placeholder="Search..."
    >
      <Option value="1">Option 1</Option>
      <Option value="2">Option 2</Option>
      <Option value="3">Option 3</Option>
    </Select>
  );
};

export default DebouncedSelect;

    Debounce with Hoc

    separate the debounce logic and the HOC into separate files for better organization

    debounce.js
    // debounce.js
const debounce = (func, delay) => {
    let timeoutId;
    return function (...args) {
        clearTimeout(timeoutId);
        timeoutId = setTimeout(() => func.apply(this, args), delay);
    };
};

export default debounce;
    withDebounce.js
    import React from 'react';
import debounce from './debounce';

const withDebounce = (WrappedComponent) => {
    return class WithDebounce extends React.Component {
        debouncedFunction = debounce((...args) => {
            console.log('Debounced function called with args:', args);
        }, 500); // 500ms debounce delay

        render() {
            return (
                <WrappedComponent
                    {...this.props}
                    onDebouncedAction={this.debouncedFunction}
                />
            );
        }
    };
};

export default withDebounce;
    MyComponent.js
    import React from 'react';
import withDebounce from './withDebounce';

const MyComponent = ({ onDebouncedAction }) => {
    const handleClick = () => {
        onDebouncedAction('Click event');
    };

    return (
        <div>
            <button onClick={handleClick}>Click me</button>
        </div>
    );
};

export default withDebounce(MyComponent);

    Now you can import and use the MyComponent in your main application file:

    App.js
    import React from "react";
import MyComponent from "./MyComponent";

const App = () => {
  return (
    <div>
      <MyComponent />
    </div>
  );
};

export default App;

    This structure helps maintain a clean separation of concerns, making each file focused on its specific responsibility. The debounce utility is isolated in its own file, and the HOC is defined separately, promoting code modularity and reusability.
